	  function getQueryParams(){
		    try{
		        url = window.location.href;
		        query_str = url.substr(url.indexOf('?')+1, url.length-1);
		        r_params = query_str.split('&');
		        params = {}
		        for( i in r_params){
		            param = r_params[i].split('=');
		            params[ param[0] ] = param[1];
		        }
		        return params;
		    }
		    catch(e){
		       return {};
		    }
		}

function isNumberPressed(evt, maxLength){
    var charCode = (evt.which) ? evt.which : evt.keyCode;
    if (charCode > 31 && (charCode<48 || charCode>57))
        return false;
    evt = evt || window.event;
    target = (evt.target) ? evt.target : window.event.srcElement; 
    var targetValue = (target.value) ? target.value : target.textContent;
    if(targetValue != undefined && targetValue.length >= maxLength){
    	return false;
    }
    return true;
}

			var data = {};
			data["userid"] = loginUserId;
			data["userOTP"] = inputOtp;
			data["dbName"] = databaseName;
			data["ismobleOtp"] = ismobOtp;

				method : "POST",
				data : JSON.stringify(data),
				url : "/rsunify/app/validateOTP",
				dataType: "json",
				async: false,
				success : function(datObj) {
					if (datObj) {

								}
							}]

					}
				}
			});
		}
		return returnResponse;
	}
